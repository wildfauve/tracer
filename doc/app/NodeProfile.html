def create_if_instances
  if @start_inst && @end_inst
    rel = Relinstance.relfactory({:type => @rel.id})
    @start_inst.create_or_update_the_node(:rel => rel, :other_node => @end_inst)
    @end_inst.create_or_update_the_node(:rel => rel, :other_node => @start_inst)      
  end
end

def end=(profile)
  if profile.nil?
    @end_type = nil
    @end_inst = nil
    return
  end
  if profile[:type]
    @end_type = Type.find(profile[:type])
  elsif profile[:inst]
    @end_inst = Node.find(profile[:inst])
    @end_type = @end_inst.type
  else
    raise Exception
  end
end

def start=(profile)
  if profile.nil?
    @end_type = nil
    @end_inst = nil
    return nil
  end
  if profile[:type]
    @start_type = Type.find(profile[:type])
  else # its an :inst =>
    @start_inst = Node.find(profile[:inst])
    @start_type = @start_inst.type
  end
end
