def self.create_it(params)
  ts = Typeset.new.load(params)
  matx = Compare.new
  matx.x = ts.start_type.id
  matx.y = ts.end_type.id
  matx.rel = ts.rel_type.id
  matx.name = ts.standardise_name
  matx.save!
  matx
end
